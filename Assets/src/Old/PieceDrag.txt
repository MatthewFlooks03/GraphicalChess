using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PieceDrag : MonoBehaviour
{
    bool canMove = true;
    private Vector2 difference;
    private Vector2 initialPos;

    private void OnMouseDown()
    {
        initialPos = transform.position;
        if (canMove)
        {
            difference = (Vector2)CameraControl.mainCamera.ScreenToWorldPoint(Input.mousePosition) - (Vector2)transform.position;
        }
    }
    
    private void OnMouseDrag()
    {
        if (canMove)
        {
            transform.position = (Vector2)CameraControl.mainCamera.ScreenToWorldPoint(Input.mousePosition) - difference;
        }
    }

    private void OnMouseUp()
    {

        //Reset to square
        float x = (float)Math.Round(transform.position.x / GraphicalBoard.boardScaleFactor) * GraphicalBoard.boardScaleFactor;
        float y = (float)Math.Round(transform.position.y / GraphicalBoard.boardScaleFactor) * GraphicalBoard.boardScaleFactor;

        if (x < 0 || x > 7 * GraphicalBoard.boardScaleFactor || y < 0 || y > 7 * GraphicalBoard.boardScaleFactor)
        {
            transform.position = initialPos;
            return;
        }
        transform.position  = new Vector2(x, y);

        Piece.boardArray
    }

}
