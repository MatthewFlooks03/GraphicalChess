using UnityEngine;

internal class CameraControl
{
    public static Camera mainCamera;

    public static void InitializeCamera()
    {
        mainCamera = GameObject.Find("MainCamera").GetComponent<Camera>();
        mainCamera.orthographic = true;
        mainCamera.orthographicSize = GraphicalBoard.boardScaleFactor * 5f;
        mainCamera.transform.position = new Vector3(GraphicalBoard.boardScaleFactor * 3.5f, GraphicalBoard.boardScaleFactor * 3.5f, -10);
    }
}
